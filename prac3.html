#include <stdio.h>
void main() {
  char ch = 'Z'; // Where 'Z' is first letter of your first name
  printf("ch = %c\n", ch); // printing character data
  // print decimal or integer data with d and i
  int x = 45; // Assign x the integer value of your faculty id
  printf("x = %d\n", x);
  int y = 90; // Assign y the integer value of your enrollment no
  printf("y = %i\n", y);

  float f = 12.67;
  printf("f (Float Value) = %f\n", f); //print float value
  printf("f (Scientific Notation) %e\n", f); //print in scientific notation
  // print x in Octal and Hexadecimal format
  printf("x in Octal format = %o\n", x);
  printf("x in Hexadecimal format = %x\n", y);

  char str[] = "Hello Zaid!";
  printf("%s\n", str);
  // shift to the right 20 characters including the string
  printf("%20s\n", str);
  // left align
  printf("%-20s\n", str);
  // shift to the right including the string, 
  // and print string up to 5 character
  printf("%20.5s\n", str);
  // left align and print string up to 5 character
  printf("%-20.5s\n", str);

}
